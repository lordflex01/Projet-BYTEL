	INDEX `IDX_A8F11980D2235D39` (`tache_id`) USING BTREE,

	INDEX `IDX_A8F119809B0F88B1` (`activite_id`) USING BTREE,

	CONSTRAINT `FK_A8F11980D2235D39` FOREIGN KEY (`tache_id`) REFERENCES `taches` (`id`) ON UPDATE RESTRICT ON DELETE RESTRICT,
	CONSTRAINT `FK_A8F11980D780BF39` FOREIGN KEY (`codeprojet_id`) REFERENCES `code_projet` (`id`) ON UPDATE RESTRICT ON DELETE RESTRICT

ENGINE=InnoDB

;
